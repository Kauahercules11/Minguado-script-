-- Function to change text color over time
local function updateTextColor()
	local hue = 0
	while true do
		hue = hue + 0.01
		if hue > 1 then hue = 0 end
		textLabel.TextColor3 = Color3.fromHSV(hue, 1, 1)
		task.wait(0.1)
	end
end
