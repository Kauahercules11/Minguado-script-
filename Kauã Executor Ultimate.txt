-- Fun√ß√£o de output otimizada
local function AddOutput(text, color)
    if not text or string.len(text) == 0 then return end
    color = color or Color3.fromRGB(200, 200, 200)
    
    if OutputText.Text == "-- Output aparecer√° aqui --" then
        OutputText.Text = ""
    end
    
    OutputText.Text = OutputText.Text .. os.date("[%H:%M:%S] ") .. tostring(text) .. "\n"
    OutputText.TextColor3 = color
end

-- Fun√ß√£o para executar scripts longos em partes
local function ExecuteLongScript(code)
    -- Dividir o c√≥digo em partes menores
    local chunkSize = 100000 -- Tamanho de cada parte (ajuste conforme necess√°rio)
    local chunks = {}
    
    for i = 1, #code, chunkSize do
        table.insert(chunks, code:sub(i, i + chunkSize - 1))
    end
    
    -- Executar cada parte sequencialmente
    local env = {}
    local results = {}
    
    -- Configurar environment
    setmetatable(env, {__index = _G})
    
    -- Fun√ß√µes b√°sicas
    env.print = function(...)
        local args = {...}
        AddOutput(table.concat(args, "\t"), Color3.fromRGB(0, 162, 255))
    end
    
    env.warn = function(...)
        local args = {...}
        AddOutput(table.concat(args, "\t"), Color3.fromRGB(255, 165, 0))
    end
    
    env.error = function(...)
        local args = {...}
        AddOutput(table.concat(args, "\t"), Color3.fromRGB(255, 50, 50))
        error(args[1])
    end
    
    -- Executar cada parte
    for i, chunk in ipairs(chunks) do
        local fn, err = loadstring(chunk)
        if not fn then
            AddOutput("‚ùå Erro ao compilar parte "..i..": "..tostring(err), Color3.fromRGB(255, 50, 50))
            return false
        end
        
        setfenv(fn, env)
        
        local success, result = pcall(fn)
        if not success then
            AddOutput("‚ùå Erro na parte "..i..": "..tostring(result), Color3.fromRGB(255, 50, 50))
            return false
        end
        
        if result ~= nil then
            table.insert(results, result)
        end
    end
    
    return true, results
end

-- Fun√ß√£o de execu√ß√£o segura
local function ExecuteCode()
    local code = CodeTextBox.Text
    
    -- Mostra notifica√ß√£o
    StarterGui:SetCore("SendNotification", {
        Title = "Kau√£ Executor",
        Text = "Executando c√≥digo...",
        Icon = "rbxassetid://",
        Duration = 2
    })
    
    -- Limpa o output
    if OutputText.Text ~= "-- Output aparecer√° aqui --" then
        OutputText.Text = ""
    end
    
    -- Verifica se √© um script longo (mais de 500KB)
    if #code > 500000 then
        AddOutput("‚ö†Ô∏è Script muito grande detectado. Executando em partes...", Color3.fromRGB(255, 165, 0))
        
        local success, results = ExecuteLongScript(code)
        if success then
            AddOutput("‚úÖ Execu√ß√£o conclu√≠da com sucesso!", Color3.fromRGB(0, 255, 0))
            if #results > 0 then
                AddOutput("Resultados: "..tostring(#results).." valores retornados", Color3.fromRGB(180, 255, 180))
            end
        end
        return
    end
    
    -- Cria environment seguro para scripts menores
    local env = {
        print = function(...)
            local args = {...}
            AddOutput(table.concat(args, "\t"), Color3.fromRGB(0, 162, 255))
        end,
        warn = function(...)
            local args = {...}
            AddOutput(table.concat(args, "\t"), Color3.fromRGB(255, 165, 0))
        end,
        error = function(...)
            local args = {...}
            AddOutput(table.concat(args, "\t"), Color3.fromRGB(255, 50, 50))
            error(args[1])
        end,
        loadstring = function(c)
            local fn, err = loadstring(c)
            if not fn then fn, err = load(c) end
            if not fn then error(err or "Erro de compila√ß√£o") end
            return fn
        end,
        game = game,
        workspace = workspace,
        script = script,
        wait = task.wait,
        pcall = pcall,
        require = require,
        Vector3 = Vector3,
        CFrame = CFrame,
        Instance = Instance
    }
    
    -- Suporte para = comando
    if string.sub(code, 1, 1) == "=" then
        code = "print(" .. string.sub(code, 2) .. ")"
    end
    
    -- Executa o c√≥digo
    local success, result = pcall(function()
        local fn, err = env.loadstring(code)
        if not fn then error(err) end
        setfenv(fn, env)
        return fn()
    end)
    
    -- Mostra resultados
    if success then
        AddOutput("‚úÖ Execu√ß√£o conclu√≠da com sucesso!", Color3.fromRGB(0, 255, 0))
        if result ~= nil then
            AddOutput("Resultado: "..tostring(result), Color3.fromRGB(180, 255, 180))
        end
    else
        AddOutput("‚ùå Erro na execu√ß√£o:", Color3.fromRGB(255, 50, 50))
        AddOutput(tostring(result):gsub("^.+:%d+: ", ""), Color3.fromRGB(255, 150, 150))
    end
end
